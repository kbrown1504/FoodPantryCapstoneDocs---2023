import { Component, OnInit } from '@angular/core';
import { FirebaseService } from '../services/firebase.service';
import { User } from '../../../../globalModels/user';
import { ModalController } from '@ionic/angular';
import { PrivacyModalComponent } from '../privacy-modal/privacy-modal.component';
import { TutorialModalComponent } from '../tutorial-modal/tutorial-modal.component';
import { Router } from '@angular/router';
import { AuthService } from '../services/auth.service';

/**
 * Profile page
 */
@Component({
  selector: 'app-profile',
  templateUrl: './profile.page.html',
  styleUrls: ['./profile.page.scss'],
})
export class ProfilePage implements OnInit {
  user: string
  constructor(
    private firebase: FirebaseService,
    private modalController: ModalController,
    private router: Router,
    private authService: AuthService
  ) { 
    this.user =  ""
  }

  /**
   * Retrieves the current user on init
   */
  async ngOnInit() {
    let userobj = await this.authService.getCurrentUser()
    this.user = userobj.email
  }

  signOut() {
    this.authService.signOut()
    this.router.navigateByUrl("login")
  }

  async showPrivacyModal() {
    const modal = await this.modalController.create(
      {
        component: PrivacyModalComponent,
        cssClass: "modal",
        backdropDismiss: false,
      }
    )

    modal.onDidDismiss().then(() => {
      console.log("dismissed")
    })

    await modal.present()
  }

  async showTutorialModal(){
    const modal = await this.modalController.create(
      {
        component: TutorialModalComponent,
        cssClass: "modal",
        backdropDismiss: false
      }
    )

    modal.onDidDismiss().then(() => {
      console.log("dismissed")
    })

    await modal.present()
  }
}
