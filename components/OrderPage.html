import { Component, OnInit } from '@angular/core';
import { ModalController } from '@ionic/angular';
import { observable, Observable } from 'rxjs';
import { FoodItem } from '../../../../globalModels/foodItem';
import { OrderModalComponent } from '../order-modal/order-modal.component';
import { FirebaseService } from '../services/firebase.service';
import { Order } from '../../../../globalModels/order';
import { Router } from '@angular/router';
import { AlertService } from '../services/alert.service';

/**
 * Order Page
 * Will display inventory and allow users to place orders
 */
@Component({
  selector: 'app-order',
  templateUrl: './order.page.html',
  styleUrls: ['./order.page.scss'],
})
export class OrderPage implements OnInit {

  orderItems: FoodItem[] = []
  foodItems: FoodItem[] = []

  constructor(
    private modalController: ModalController,
    private firebase: FirebaseService,
    private router: Router,
    private alertService: AlertService
  ) {
    this.foodItems = [
      // {
      //   id: 'apple1',
      //   name: 'Apple1',
      //   quantity: 10,
      //   imageUrl: 'https://thumbs.dreamstime.com/b/red-apple-half-red-apple-16354511.jpg'
      // },
      // {
      //   id: 'apple2',
      //   name: 'Apple2',
      //   quantity: 10,
      //   imageUrl: 'https://thumbs.dreamstime.com/b/red-apple-half-red-apple-16354511.jpg'
      // }
    ]
  }

  /**
   * Retrieves inventory and process it for display
   */
  async ngOnInit() {
    try {
      this.foodItems = await this.firebase.getInventory()
      this.foodItems = this.processInventory(this.foodItems)
    } catch {
      this.alertService.presentAlert('Error', 'Could not retrieve the inventory. Please try again later.', ['Ok'])
    }

  }

  /**
   * Displays a modal with a FoodItem
   * @param item FoodItem to display in modal
   */
  async showModal(item: FoodItem) {
    const modal = await this.modalController.create(
      {
        component: OrderModalComponent,
        componentProps: {
          item: item
        },
        cssClass: "modal",
        backdropDismiss: false,
      },
    )

    modal.onDidDismiss().then((data) => {
      if (data.data != undefined) {
        let index = this.orderItems.indexOf(data.data)
        if (index >= 0) {
          this.orderItems[index].orderQuantity = data.data.orderQuantity
          if (data.data.orderQuantity == 0) {
            this.orderItems.splice(index, 1)
          }
        } 
        else {
          this.orderItems.push(data.data as FoodItem)
          console.log(data.data)
        }
      }
    })

    await modal.present()
    console.log("done")
  }

  /**
   * Collects all order items and places the order to the DB
   */
  async placeOrder() {
    try {
      let orderItems: FoodItem[] = []
      for (let item of this.foodItems) {
        if (item.orderQuantity != undefined && item.orderQuantity > 0) {
          orderItems.push(item)
        }
      }
      let orderToPlace = new Order('', new Date(), 1, 'testUser', orderItems)
      this.firebase.placeOrder(orderToPlace).then(() => {
        this.router.navigateByUrl('order-confirm')
      })
    } catch {
      this.alertService.presentAlert('Error', 'Could not place your order. Please try again later.', ['Ok'])
    }

  }

  /**
   * Removes all items from an array of FoodItems that have a quantity of 0
   * @param inven Array of FoodItems
   * @returns An Array of FoodItems that have a quantity > 0
   */
  processInventory(inven: FoodItem[]) {
    for (let i = 0; i < inven.length; i++) {
      if (inven[i].quantity <= 0) {
        inven.splice(i, 1)
      }
    }
    return inven
  }

}
