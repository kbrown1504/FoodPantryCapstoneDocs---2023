import { Component } from '@angular/core';
import { FirebaseService } from '../services/firebase.service';
import { Order } from "../../../../globalModels/order"
import { OrderStatus } from '../../../../globalModels/orderStatus';
import { Router } from '@angular/router';
import { AlertService } from '../services/alert.service';

/**
 * Home page for the user app
 * Will display User's orders and the button for placing a new one
 */
@Component({
  selector: 'app-home',
  templateUrl: 'home.page.html',
  styleUrls: ['home.page.scss'],
})
export class HomePage {

  refreshOrders = true
  orders: Order[] = []
  logoUrl: string = ""

  constructor(
    private firebase: FirebaseService,
    private router: Router,
    private alertService: AlertService
  ) {

  }

  ngOnInit() {

  }

  /**
   * Retrieves user orders and the pantry
   */
  async ionViewWillEnter() {
    try {
      if (this.refreshOrders) {
        this.orders = await this.firebase.getUserOrders()
        this.refreshOrders = false
      }
      let pantry = await this.firebase.getPantry()
      this.logoUrl = pantry.logoUrl
    }
    catch {
      this.alertService.presentAlert('Error', 'Your orders could not be retrieved! Please try again later.', ['Ok'])
    }
  }


  goToProfile() {
    this.router.navigateByUrl('profile')
  }

  goToOrder() {
    this.refreshOrders = true
    this.router.navigateByUrl('order')
  }

  toOrderDetail(order: Order) {
    let navExtras = {
      state: {
        order: order
      }
    }
    this.router.navigate(['order-detail'], navExtras)
  }

}
