import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { FirebaseService } from '../services/firebase.service';
import { AuthService } from '../services/auth.service';
import { Pantry } from '../../../../globalModels/pantry';
import { AlertService } from '../services/alert.service';

/**
 * Account creation page
 */
@Component({
  selector: 'app-signup',
  templateUrl: './signup.page.html',
  styleUrls: ['./signup.page.scss'],
})
export class SignupPage implements OnInit {
  public email: string  = ''
  public password: string = ''
  public passwordConf: string = ''

  public pantries: Pantry[] = []
  public pantry: string = ''
  constructor(
    private router: Router,
    private authService: AuthService,
    private firebaseService: FirebaseService,
    private alertService: AlertService
  ) { }

  /**
   * Retrieves pantry options from DB
   */
  async ngOnInit() {
    this.pantries = await this.firebaseService.getPantries()
  }

  /**
   * Updates the user's pantry selection
   * @param event event emitted by ion-select
   */
  setPantrySelection(event: any) {
    this.pantry = event.detail.value
  }

  /**
   * Initiates account creation.
   */
  async signUpButton() {
    if (this.password == this.passwordConf && this.pantry.length > 0) {
      await this.authService.createAccount(this.email, this.password).then(async () => {
        sessionStorage.setItem('pantryId', this.pantry)
        this.firebaseService.pantryName = this.pantry
        let user = await this.authService.getCurrentUser()
        sessionStorage.setItem('uid', user.uid)
        this.firebaseService.userId = user.uid
        this.router.navigateByUrl('home')
      }).catch(async (err) => {
        await this.alertService.presentAlert('Account Creation Failed', 'Please try again.', ['Ok'])
      })
    } else {
      if (this.pantry.length <= 0) {
        await this.alertService.presentAlert('No Pantry Selected', 'You must select a pantry before creating an account.', ['Ok'])
      } else {
        await this.alertService.presentAlert('Error', 'Passwords do not match!', ['Ok'])
      }
    }
  }

  toSignin() {
    this.router.navigateByUrl("login")
  }

}
