import { Injectable } from '@angular/core';

import { getFirestore, addDoc, getDocs, doc, DocumentSnapshot, query, where, getDoc, DocumentData, Timestamp, collection, setDoc, updateDoc } from 'firebase/firestore'

import { FoodItem } from '../../../../globalModels/foodItem'
import { Order } from '../../../../globalModels/order'
import { Pantry } from '../../../../globalModels/pantry'

const db = getFirestore()

@Injectable({
  providedIn: 'root'
})
export class FirebaseService {

  //This will be used to determine which pantry instance is accessed
  //TODO: update/change this when the user logs in
  pantryName = 'testPantry'
  userId = 'testUser'

  constructor() {

  }

  async getPantry(): Promise<Pantry> {
    let snapshot = undefined;
    try {
      snapshot = await getDoc(doc(collection(db, 'pantries/' + this.pantryName)))
    } catch (error) {
      throw (error)
    }
    if (snapshot == undefined) {
      throw Error('Food Pantry does not exist')
    }

    return this.pantryConv(snapshot)

  }

  pantryConv(snapshot: DocumentSnapshot) {
    let data = snapshot.data()!
    return new Pantry(data['name'], data['description'], data['address'], data['logoUrl'], data['primaryColor'], data['secondaryColor'], data['orderLimit'])
  }

  async getInventory(): Promise<FoodItem[]> {
    let snapshot = undefined;
    try {
      snapshot = await getDocs(collection(db, "pantries/" + this.pantryName + "/inventory/"))
    } catch (error) {
      //throw error to outer page
      throw error
    }
    if (snapshot == undefined) {
      throw Error('No Data Found')
    }
    
    let toReturn: FoodItem[] = []
    snapshot.forEach((doc) => {
      if (doc.exists()) {
        toReturn.push(this.foodItemConv(doc))
      }
    })
    return toReturn
  }

  foodItemConv(snap: DocumentSnapshot) {
    let docData = snap.data()!
    return new FoodItem(snap.id, docData['name'], docData['quantity'], docData['orderQuan'], docData['limit'], docData['imageUrl'])
  }

  //TODO/To figure out if we need: Have this method also update the pantry inventory when the user places an order??????
  async placeOrder(order: Order) {
    try {
      //Create Document
      let docRef = await addDoc(collection(db, 'pantries/' + this.pantryName + '/orders/'), {
        date: Timestamp.fromDate(order.timestamp),
        status: 1,
        uid: this.userId
      })
      //Add Items to Order's items collection
      let col = collection(db, docRef.path + '/items')
      for (let item of order.items) {
        setDoc(doc(col, item.id), {
          name: item.name,
          quantity: item.orderQuantity
        })
      }

      //Update User Order Array
      let userRef = doc(db, 'pantries/' + this.pantryName + '/users/' + this.userId)
      let user = await getDoc(userRef)
      if (user.exists()) {
        let orders = user.data()['userOrders']
        orders.push(docRef)
        await updateDoc(userRef, {
          userOrders: orders
        })
      } else {
        throw Error('User does not exist')
      }
      
    } catch (err) {
      throw err
    }
  }

  async getUserOrders(): Promise<Order[]> {
    let snapshot = undefined
    let userDoc = undefined
    try {
      let usersRef = collection(db, 'pantries/' + this.pantryName + '/users/')
      let q = query(usersRef, where('uid', '==', this.userId))
      snapshot = await getDocs(q)
      if (snapshot.docs.length > 1) {
        //This should never happen
        throw Error('More than one user found!')
      } else if (snapshot.docs.length < 1){
        throw Error('No user found!')
      } else {
        userDoc = snapshot.docs[0]
      }
    } catch (err) {
      throw err
    }
    
    let toReturn = []
    let orders = userDoc.data()['userOrders']
    for (let orderRef of orders) {
      let document: DocumentSnapshot<DocumentData> = await getDoc(orderRef)
      let orderToAdd: Order = await this.orderConv(document)
      toReturn.push(orderToAdd)
    }

    return toReturn
  }

  async orderConv(snap: DocumentSnapshot) {
    let docData = snap.data()!
    if (docData == undefined) {
      throw Error('Data does not exist!')
    }
    let items: FoodItem[] = []
    //TODO: actually pull for items for order
    items = await this.getOrderItems(snap)
    return new Order(snap.id, docData['date'], docData['status'], docData['uid'], items)
  }

  async getOrderItems(order: DocumentSnapshot): Promise<FoodItem[]> {
    let snapshot = undefined
    let orderPath = order.ref.path
    try {
      snapshot = await getDocs(collection(db, orderPath + '/items'))
    } catch (error) {
      throw error
    }
    if (snapshot == undefined) {
      throw Error('No Data Found')
    }

    let toReturn: FoodItem[] = []
    snapshot.forEach((item) => {
      toReturn.push(this.foodItemConv(item))
    })
    return toReturn
  }

  async cancelOrder(orderToCancel: Order) {
    let docId = orderToCancel.id
    let docRef = doc(db, 'pantries/' + this.pantryName + '/orders/' + docId)
    let snapshot = await getDoc(docRef)
    if (snapshot.exists()) {
      await updateDoc(docRef, {
        //Status 5 corresponds with Order Canceled
        status: 5
      })
    } else {
      throw Error('Order does not exist!')
    }
  }

}
