import { Injectable } from '@angular/core';

import { getFirestore, addDoc, getDocs, doc, DocumentSnapshot, query, where, getDoc, DocumentData, Timestamp, collection, setDoc, updateDoc } from 'firebase/firestore'

import { FoodItem } from '../../../../globalModels/foodItem'
import { Order } from '../../../../globalModels/order'
import { Pantry } from '../../../../globalModels/pantry'
import { LoadingService } from './loading.service';

const db = getFirestore()

@Injectable({
  providedIn: 'root'
})
export class FirebaseService {

  //This will be used to determine which pantry instance is accessed
  //TODO: update/change this when the user logs in
  pantryName: string = 'testPantry'
  userId: string = 'testUser'

  constructor(
    private loadingService: LoadingService,
  ) {
    let temp = sessionStorage.getItem('uid')
    let temp2 = sessionStorage.getItem('pantryId')
    if (temp != null && temp2 != null) {
      this.userId = temp
      this.pantryName = temp2
    }
  }

  /**
   * Retrieves all pantries
   * @returns Promise containing an array of pantries in the database
   */
  async getPantries(): Promise<Pantry[]> {
    await this.loadingService.showLoader()
    let snapshot = undefined
    try {
      snapshot = await getDocs(collection(db, 'pantries'))
    } catch (error) {
      this.loadingService.dismissLoader()
      throw error
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('No Pantries Found')
    }
    var toReturn: Pantry[] = []
    snapshot.forEach((pantry) => {
      toReturn.push(this.pantryConv(pantry))
    })
    this.loadingService.dismissLoader()
    return toReturn
  }

  /**
   * Retrives the current pantry
   * @returns A promise containing the pantry object from the DB
   */
  async getPantry(): Promise<Pantry> {
    await this.loadingService.showLoader()
    let snapshot = undefined;
    try {
      snapshot = await getDoc(doc(db, 'pantries/' + this.pantryName))
    } catch (error) {
      this.loadingService.dismissLoader()
      throw (error)
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('Food Pantry does not exist')
    }
    this.loadingService.dismissLoader()
    return this.pantryConv(snapshot)

  }

  /**
   * Converts the firebase pantry dictionary to a custom Pantry object
   * @param snapshot Firebase document snapshot to convert
   * @returns A Pantry object
   */
  pantryConv(snapshot: DocumentSnapshot) {
    let data = snapshot.data()!
    return new Pantry(snapshot.id, data['name'], data['description'], data['address'], data['logoUrl'], data['primaryColor'], data['secondaryColor'], data['orderLimit'])
  }

  /**
   * Retrieves the item in the inventory for the current pantry
   * @returns A promise containing an array of items in the invetory
   */
  async getInventory(): Promise<FoodItem[]> {
    await this.loadingService.showLoader()
    let snapshot = undefined;
    try {
      snapshot = await getDocs(collection(db, "pantries/" + this.pantryName + "/inventory/"))
    } catch (error) {
      this.loadingService.dismissLoader()
      //throw error to outer page
      throw error
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('No Data Found')
    }
    
    let toReturn: FoodItem[] = []
    snapshot.forEach((doc) => {
      if (doc.exists()) {
        toReturn.push(this.foodItemConv(doc))
      }
    })
    this.loadingService.dismissLoader()
    return toReturn
  }

  /**
   * Converts the firebase item document to a FoodItem object
   * @param snap Firebase docuement snapshot to convert
   * @returns a FoodItem object
   */
  foodItemConv(snap: DocumentSnapshot) {
    let docData = snap.data()!
    return new FoodItem(snap.id, docData['name'], docData['quantity'], docData['orderQuan'], docData['limit'], docData['imageUrl'])
  }

  /**
   * places an order for a user. Also updates the invetory based on what the user orders
   * @param order Order to be placed
   */
  async placeOrder(order: Order) {
    await this.loadingService.showLoader()
    try {
      //Create Document
      let docRef = await addDoc(collection(db, 'pantries/' + this.pantryName + '/orders/'), {
        date: Timestamp.fromDate(order.timestamp),
        status: 1,
        uid: this.userId
      })
      //Add Items to Order's items collection
      let col = collection(db, docRef.path + '/items')
      for (let item of order.items) {
        setDoc(doc(col, item.id), {
          name: item.name,
          quantity: item.orderQuantity
        })
        let newQuan = item.quantity - item.orderQuantity!
        console.log(item.id)
        console.log(newQuan)
        updateDoc(doc(db, 'pantries/' + this.pantryName + '/inventory/' +item.id), {
          quantity: newQuan
        })
      }

      //Update User Order Array
      let userRef = doc(db, 'pantries/' + this.pantryName + '/users/' + this.userId)
      let user = await getDoc(userRef)
      if (user.exists()) {
        let orders = user.data()['userOrders']
        orders.push(docRef)
        await updateDoc(userRef, {
          userOrders: orders
        })
      } else {
        this.loadingService.dismissLoader()
        throw Error('User does not exist')
      }
      
    } catch (err) {
      this.loadingService.dismissLoader()
      throw err
    }
    this.loadingService.dismissLoader()
  }

  /**
   * Retrieves all orders for the currently logged in user
   * @returns A promise containing an array of Orders
   */
  async getUserOrders(): Promise<Order[]> {
    await this.loadingService.showLoader()
    let snapshot = undefined
    let userDoc = undefined
    try {
      let usersRef = collection(db, 'pantries/' + this.pantryName + '/users/')
      let q = query(usersRef, where('uid', '==', this.userId))
      snapshot = await getDocs(q)
      if (snapshot.docs.length > 1) {
        //This should never happen
        this.loadingService.dismissLoader()
        throw Error('More than one user found!')
      } else if (snapshot.docs.length < 1){
        this.loadingService.dismissLoader()
        throw Error('No user found!')
      } else {
        userDoc = snapshot.docs[0]
      }
    } catch (err) {
      this.loadingService.dismissLoader()
      throw err
    }
    
    let toReturn = []
    let orders = userDoc.data()['userOrders']
    for (let orderRef of orders) {
      // console.log('get order')
      let document: DocumentSnapshot<DocumentData> = await getDoc(orderRef)
      // console.log('got order')
      let orderToAdd: Order = await this.orderConv(document)
      toReturn.push(orderToAdd)
    }
    this.loadingService.dismissLoader()
    return toReturn
  }

  /**
   * Converts firebase document to and Order object
   * @param snap firebase document snapshot to convert
   * @returns an Order object
   */
  async orderConv(snap: DocumentSnapshot) {
    let docData = snap.data()!
    if (docData == undefined) {
      throw Error('Data does not exist!')
    }
    let items: FoodItem[] = []
    //TODO: actually pull for items for order
    items = await this.getOrderItems(snap)
    return new Order(snap.id, docData['date'], docData['status'], docData['uid'], items)
  }

  /**
   * Retreives the items contained within an order
   * @param order firebase document snapshot to get items from
   * @returns A promise containing and array of FoodItems in an order
   */
  async getOrderItems(order: DocumentSnapshot): Promise<FoodItem[]> {
    let snapshot = undefined
    let orderPath = order.ref.path
    try {
      snapshot = await getDocs(collection(db, orderPath + '/items'))
    } catch (error) {
      throw error
    }
    if (snapshot == undefined) {
      throw Error('No Data Found')
    }

    let toReturn: FoodItem[] = []
    snapshot.forEach((item) => {
      toReturn.push(this.foodItemConv(item))
    })
    return toReturn
  }

  /**
   * Cancels and order. Also adds the items from the order back into the inventory
   * @param orderToCancel
   */
  async cancelOrder(orderToCancel: Order) {
    await this.loadingService.showLoader()
    let docId = orderToCancel.id
    let docRef = doc(db, 'pantries/' + this.pantryName + '/orders/' + docId)
    let snapshot = await getDoc(docRef)
    if (snapshot.exists()) {
      await updateDoc(docRef, {
        //Status 5 corresponds with Order Canceled
        status: 5
      })
      for (let item of orderToCancel.items) {
        let itemRef = doc(db, 'pantries/' + this.pantryName + '/inventory/' + item.id)
        let invenItem = await getDoc(itemRef)
        let invenItemData = invenItem.data()
        if (invenItem.exists()) {
          let currentQuan = invenItemData!['quantity']
          let newQuan = currentQuan + item.quantity
          await updateDoc(itemRef, {
            quantity: newQuan
          })
        }
      }
    } else {
      this.loadingService.dismissLoader()
      throw Error('Order does not exist!')
    }
    this.loadingService.dismissLoader()
  }

  /**
   * Creates a proble for a user in the database.
   * @param uid UID of profile to create
   * @param email email for profile
   * @param timestamp timestamp of when the account was created
   */
  async createUserProfile(uid: string, email: string, timestamp: Date) {
    try {
      let docRef = await setDoc(doc(db, 'pantries/' + this.pantryName + '/users/' + uid), {
        date: Timestamp.fromDate(timestamp),
        uid: uid,
        email: email,
        userOrders: []
      })
    } catch (err) {
      throw err;
    }
  }

}
