import { Injectable } from '@angular/core';
import { 
  getAuth,
  onAuthStateChanged,
  signOut, 
  signInWithEmailAndPassword, 
  createUserWithEmailAndPassword,
  sendEmailVerification
} from "firebase/auth"
import { LoadingService } from './loading.service';
import { FirebaseService } from './firebase.service';
import { AlertService } from './alert.service';

const auth = getAuth()
onAuthStateChanged(auth, (user) => {
  if (user) {
    //user is signed in
    const uid = user.uid
  } else {
    //No user signed in
  }
})

@Injectable({
  providedIn: 'root'
})
export class AuthService {


  constructor(
    private loadingService: LoadingService,
    private firebaseService: FirebaseService,
    private alertService: AlertService
    ) {

   }

   /**
    * Retrieves the current user
    * @returns Promise containing a firebase user object
    */
   async getCurrentUser(): Promise<any> {
    return new Promise(async (resolve, reject) => {
      const unsub = await auth.onAuthStateChanged(user => {
        resolve(user)
      }, reject)
    })
  }

  /**
   * Signs a user in with email and password
   * @param email 
   * @param password 
   */
   async signIn(email: string, password: string) {
    await this.loadingService.showLoader()
    await signInWithEmailAndPassword(auth, email, password).catch((err) => {
      this.loadingService.dismissLoader()
      this.alertService.presentAlert('Login Failed', 'Please try again.', ['Ok'])
      // throw err
    })
    this.loadingService.dismissLoader()
   }

   /**
    * Creates a new account with an email and password. Also signs the user in
    * @param email 
    * @param password 
    */
   async createAccount(email:string, password: string) {
    await this.loadingService.showLoader()
    await createUserWithEmailAndPassword(auth, email, password).then(async (user) => {
      await this.firebaseService.createUserProfile(user.user.uid, user.user.email!, new Date())
    }).catch((err) => {
      this.loadingService.dismissLoader()
      //this.alertService.presentAlert('Account Creation Failed', 'Please try again.', ['Ok'])
      // throw err
    })
    this.loadingService.dismissLoader()
   }

   /**
    * Signs the user out
    */
   signOut() {
    localStorage.removeItem('uid')
    localStorage.removeItem('pantryId')
    signOut(auth)
   }

   /**
    * Checks if the current user has verified their email
    * @returns Promise containing true if the email is verified, false otherwise
    */
   async isEmailVerififed(): Promise<boolean> {
    let user = await this.getCurrentUser()
    return user.isEmailVerififed
   }

   /**
    * Sends a verification email to the current user
    */
   async sendVerificationEmail() {
    await this.loadingService.showLoader()
    let user = await this.getCurrentUser()
    sendEmailVerification(user).catch((err) => {
      this.loadingService.dismissLoader()
      throw err
    })
    this.loadingService.dismissLoader()
   }


}
